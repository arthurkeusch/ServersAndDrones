151 Matrix33 mat;
152 Vector2D A = tabPts[0];
153 Vector2D B = tabPts[1];
154 Vector2D C = tabPts[2];
155 mat.m[0][0] = (A.x - M.x);
156 mat.m[0][1] = (A.y - M.y);
157 mat.m[0][2] = (A.x*A.x-M.x*M.x)+(A.y*A.y-M.y*M.y);
158 mat.m[1][0] = (B.x - M.x);
159 mat.m[1][1] = (B.y - M.y);

161 mat.m[2][0] = (C.x - M.x);
162 mat.m[2][1] = (C.y - M.y);
163 mat.m[2][2] = (C.x*C.x-M.x*M.x)+(C.y*C.y-M.y*M.y);
